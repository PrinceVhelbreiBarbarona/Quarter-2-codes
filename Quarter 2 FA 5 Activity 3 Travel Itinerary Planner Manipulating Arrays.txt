print("Welcome to your Travel Itinerary Planner!")

destinations = []

for i in range(1, 6):
    dest = input(f"Enter destination #{i}: ")
    destinations.append(dest)

print("Your initial itinerary:")
for i, place in enumerate(destinations, start=1):
    print(f"{i}. {place}")

print("Let's update your 2nd and 5th destinations in your itinerary.")

new_second = input("Enter a new destination for spot #2: ")
destinations[1] = new_second 

new_fifth = input("Enter a new destination for spot #5: ")
destinations[4] = new_fifth  

print("Your updated itinerary:")
for i, place in enumerate(destinations, start=1):
    print(f"{i}. {place}")

print("Your itinerary has been updated successfully!")
